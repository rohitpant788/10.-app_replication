{{- if .Values.ingress.enabled }}

2. Port-forward the ingress controller (run as Administrator):
{{- if .Values.ingress.tls.enabled }}
   kubectl port-forward -n ingress-nginx svc/ingress-nginx-controller 80:80 443:443
{{- else }}
   kubectl port-forward -n ingress-nginx svc/ingress-nginx-controller 80:80
{{- end }}

3. Access the application:
{{- if .Values.ingress.tls.enabled }}

   {{- end }}
   {{- if .Values.refdataService.enabled }}
   - RefData API: https://{{ .Values.ingress.host }}/refdata/countries
   {{- end }}
   {{- if .Values.searchService.enabled }}
   - Search API: https://{{ .Values.ingress.host }}/search/cases
   {{- end }}
   {{- if .Values.fileService.enabled }}
   - File API: https://{{ .Values.ingress.host }}/file/*
   {{- end }}
